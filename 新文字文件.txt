function checkSearch() {
	var frm = arguments[0];

	var checkNoValue = function() {
		for (var i = 0; i < arguments["length"]; i++) {
			var value = arguments[i];
			if (value != "" &&
				value != undefined)
				return false;
		}
		return true;
	};
	
	if (checkNoValue(
		frm["orgName"]["value"],
		frm["orgId"]["value"],
		frm["tenderName"]["value"],
		frm["tenderId"]["value"],
		frm["tenderWay"]["value"],
		frm["tenderStartDate"]["value"],
		frm["tenderEndDate"]["value"], 
		frm["proctrgCate"]["value"])) {
		return showCheckFailure("因資料量太大 , 請多輸入1個查詢條件");
	}
	
	var dateRange; // 日期區間限制
	
	if(checkNoValue(frm["tenderId"]["value"])){
		if (checkNoValue(frm["tenderStartDate"]["value"])) {
			return showCheckFailure("招標公告起日為必填");
		}
		if (checkNoValue(frm["tenderEndDate"]["value"])) {
			return showCheckFailure("招標公告迄日為必填");
		}
		dateRange = 93;
	}
		
	if ( !checkSearchDate(frm,
		"tenderStartDate", "tenderEndDate", "公告日期",dateRange) )
		return false;
	return true;
}

function changeToAdvance(){
	
	var frm = arguments[0];
	
	var checkNoValue = function() {
		for (var i = 0; i < arguments["length"]; i++) {
			var value = arguments[i];
			if (value != "" &&
				value != undefined)
				return false;
		}
		return true;
	};
	
	if (checkNoValue(
		frm["orgId"]["value"],		
		frm["tenderId"]["value"],
		frm["tenderWay"]["value"],
		frm["tenderStartDate"]["value"],
		frm["tenderEndDate"]["value"],
		frm["proctrgCate"]["value"])) {
		
		if (!checkNoValue(
			frm["orgName"]["value"],
			frm["tenderName"]["value"])) {
				return true;		
		}else{
			return false;
		}
	}else{
		return false;
	}	
}

function formSubmit() {
	var frm = document["forms"]["TenderActionForm"];
	
	if (document.getElementById("tenderType").value == 'searchAppeal'){
		goSearchAppeal();
		return false;
	}

	if (document.getElementById("tenderType").value == 'publicRead'){
		goPublicRead();
		return false;
	}	
	
	try {

		var loc = document.getElementById("selLocation");		
		
		if (loc != undefined && loc != null)
			frm["location"]["value"] = loc.location_pk == undefined ?
				"" : loc.location_pk + "";
	
		if (!checkSearch(frm))
			return false;
		if(!$('#rangeTenderDateRadio').attr('checked')){
			$('#isSpdt').val("N");
		}else{
			$('#isSpdt').val("Y");
		}
		if (changeToAdvance(frm)){
			frm.action = "./tender.do?searchMode=common&searchType=advance&searchTarget=TPAM";
		}else{

			frm.action = "./tender.do?searchMode=common&searchType=basic";
		}		
		
		frm.submit();
	}
	catch (e) {
		if (e["description"] != undefined)
			alert(e["description"]);
		else
			alert(e);
	}
}
